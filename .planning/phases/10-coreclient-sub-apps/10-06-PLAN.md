def test_dns_zone_list_success(runner):
    with patch("bunny_cdn_sdk.core.CoreClient") as MockClient:
        MockClient.return_value.iter_dns_zones.return_value = iter([DNS_ZONE])
        result = runner.invoke(app, ["--api-key", "k", "dns-zone", "list"])
    assert result.exit_code == 0
    assert "example.com" in result.output

def test_dns_zone_list_error(runner):
    with patch("bunny_cdn_sdk.core.CoreClient") as MockClient:
        MockClient.return_value.iter_dns_zones.side_effect = _mock_exc(BunnyAuthenticationError, 401)
        result = runner.invoke(app, ["--api-key", "k", "dns-zone", "list"])
    assert result.exit_code == 1

def test_dns_zone_list_json(runner):
    with patch("bunny_cdn_sdk.core.CoreClient") as MockClient:
        MockClient.return_value.iter_dns_zones.return_value = iter([DNS_ZONE])
        result = runner.invoke(app, ["--api-key", "k", "--json", "dns-zone", "list"])
    assert result.exit_code == 0
    parsed = json.loads(result.output)
    assert isinstance(parsed, list)
    assert parsed[0]["Domain"] == "example.com"

from __future__ import annotations

from bunny_cdn_sdk._exceptions import BunnyNotFoundError
from bunny_cdn_sdk.cli import app

    return cls(status_code=status_code, message=message, response=MagicMock())

def test_purge_success(runner):
    with patch("bunny_cdn_sdk.core.CoreClient") as MockClient:
        MockClient.return_value.purge_url.return_value = {}

def test_purge_missing_auth(runner):
