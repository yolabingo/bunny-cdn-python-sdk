```python
"""Retry transport for httpx — exponential backoff with jitter (internal)."""

from __future__ import annotations

import asyncio
import random
from datetime import datetime, timezone
from email.utils import parsedate_to_datetime

import httpx


def _parse_retry_after(value: str) -> float:
    """Parse a Retry-After header value and return delay in seconds (>= 0.0).

    Supports two RFC 7231 formats:
    - Integer seconds: ``Retry-After: 60``
    - HTTP-date:       ``Retry-After: Wed, 21 Oct 2015 07:28:00 GMT``

    Returns 0.0 if the value cannot be parsed.
    """
    try:
        return max(0.0, float(value))
    except ValueError:
        try:
            retry_date = parsedate_to_datetime(value)
            delta = (retry_date - datetime.now(timezone.utc)).total_seconds()
            return max(0.0, delta)
        except Exception:  # noqa: BLE001
            return 0.0


class RetryTransport(httpx.AsyncBaseTransport):
    """Composable httpx async transport that adds retry logic to any inner transport.

    Wraps an existing ``httpx.AsyncBaseTransport`` and retries failed requests
    using exponential backoff with full jitter.

    Retry triggers:
    - HTTP 429 — respects ``Retry-After`` header when present
    - HTTP 5xx — any server error response
    - ``httpx.ConnectError`` — network connection failure
    - ``httpx.TimeoutException`` — request or connect timeout

    Usage::

        import httpx
        from bunny_cdn_sdk import RetryTransport, CoreClient

        inner = httpx.AsyncHTTPTransport()
        retry_transport = RetryTransport(inner, max_retries=3, backoff_base=0.5)
        async_client = httpx.AsyncClient(transport=retry_transport)
        core = CoreClient(api_key="...", client=async_client)
    """

    def __init__(
        self,
        inner: httpx.AsyncBaseTransport,
        *,
        max_retries: int = 3,
        backoff_base: float = 0.5,
    ) -> None:
        """Initialise the retry transport.

        Args:
            inner: The underlying transport to delegate requests to.
            max_retries: Maximum number of retries (not counting the initial attempt).
                         ``max_retries=3`` means up to 4 total calls.
            backoff_base: Base delay in seconds for exponential backoff.
                          Delay for attempt N is ``uniform(0, min(60, base * 2**N))``.
        """
        self._inner = inner
        self._max_retries = max_retries
        self._backoff_base = backoff_base

    async def handle_async_request(
        self,
        request: httpx.Request,
    ) -> httpx.Response:
        """Handle a single async request, retrying on retriable failures.

        On the final attempt (attempt == max_retries), returns the failed response
        for status-code errors (so ``_BaseClient._request`` can call
        ``raise_for_status()`` and map it to the correct SDK exception). For network
        exceptions on the final attempt, re-raises the exception directly.

        Args:
            request: The outbound httpx.Request to execute.

        Returns:
            The httpx.Response from the inner transport (possibly a non-2xx response
            on final attempt if all retries are exhausted for a status-code trigger).

        Raises:
            httpx.ConnectError: If all attempts fail with a connection error.
            httpx.TimeoutException: If all attempts fail with a timeout.
        """
        # Initialize response before loop to satisfy ty's possibly-unresolved-reference rule.
        response: httpx.Response = httpx.Response(0)

        for attempt in range(self._max_retries + 1):
            if attempt > 0:
                delay = self._backoff_delay(attempt - 1, response)
                await asyncio.sleep(delay)
            try:
                response = await self._inner.handle_async_request(request)
                if self._should_retry_response(response) and attempt < self._max_retries:
                    continue
                return response
            except (httpx.ConnectError, httpx.TimeoutException):
                if attempt >= self._max_retries:
                    raise
        # Unreachable: the loop always returns or raises before reaching here.
        return response  # type: ignore[return-value]

    async def aclose(self) -> None:
        """Close the inner transport and release its resources."""
        await self._inner.aclose()

    def _should_retry_response(self, response: httpx.Response) -> bool:
        """Return True if the response status code warrants a retry."""
        return response.status_code == 429 or 500 <= response.status_code < 600

    def _backoff_delay(
        self,
        attempt: int,
        response: httpx.Response | None = None,
    ) -> float:
        """Compute the delay in seconds before the next retry attempt.

        For 429 responses with a ``Retry-After`` header, uses the header value.
        Otherwise applies full-jitter exponential backoff:
        ``uniform(0, min(60.0, backoff_base * 2**attempt))``.

        Args:
            attempt: Zero-indexed retry attempt number (0 = first retry).
            response: The last response received, or None for network exceptions.

        Returns:
            Delay in seconds (>= 0.0).
        """
        if response is not None and response.status_code == 429:
            retry_after = response.headers.get("retry-after")
            if retry_after is not None:
                return _parse_retry_after(retry_after)
        cap = 60.0
        return random.uniform(0, min(cap, self._backoff_base * (2**attempt)))
```

Notes on the implementation:
- `response: httpx.Response = httpx.Response(0)` initialises the variable before the loop. `httpx.Response(0)` is a sentinel with status 0 that is never returned to callers. This avoids `ty`'s `possibly-unresolved-reference` error (`[tool.ty.rules] possibly-unresolved-reference = "error"` in pyproject.toml).
- `except Exception:  # noqa: BLE001` in `_parse_retry_after` — BLE001 is "blind exception"; the noqa is required because `parsedate_to_datetime` raises undocumented exceptions for malformed dates and we want the fallback behaviour. Ruff's ALL ruleset requires this annotation.
- The sentinel `response = httpx.Response(0)` is never passed to `_backoff_delay` as the 429-check argument; `_backoff_delay` is only called with the actual response from the previous attempt (stored in `response` after the first successful `handle_async_request` call).
- `import asyncio` at module level (not `from asyncio import sleep`) so tests can patch `asyncio.sleep` as the global target.
- `import random` at module level so tests can patch `bunny_cdn_sdk._retry.random.uniform` (or `random.uniform` globally) for deterministic backoff assertions.

## Threat surface

**Input validation:**
- `max_retries: int` and `backoff_base: float` accept any value including negatives. Negative `max_retries` would produce `range(negative + 1)` which is an empty range — the request is made once and returned without retry. Callers should not rely on this behaviour, but it does not create a security vulnerability.
- `Retry-After` header value comes from a remote server response. The `_parse_retry_after` function clamps the result to `max(0.0, ...)`, preventing negative delays. An attacker controlling the server could set `Retry-After: 99999999` to cause an extremely long sleep; callers should apply their own timeout at the `httpx.AsyncClient` level. The transport itself does not cap the Retry-After value (the research notes this is intentional — RETRY-03 only specifies the backoff cap for normal backoff, not for Retry-After).

**Error information leakage:**
- `RetryTransport` returns raw `httpx.Response` objects to `_BaseClient._request()`. The transport never logs or exposes response body content. No leakage risk at this layer.

**Network-level risks:**
- SSRF: `RetryTransport` does not construct URLs; it delegates unchanged `httpx.Request` objects to the inner transport. No SSRF surface introduced.
- Retry amplification: A compromised server could trigger max_retries requests per client call. Default `max_retries=3` means 4 calls max. This is an acceptable risk for the SDK's intended use case.

**Authentication concerns:**
- `RetryTransport` does not inspect, modify, or log request headers. The `AccessKey` header injected by `_BaseClient._request()` passes through opaquely. No authentication header leakage risk.
